# ============================================================
#  MPG wheel – motion layer pins (joint.N.* and axis.x.*)
#  Dual-wired so jogging works both before and after homing.
#  jog-vel-mode 0 = each encoder count moves exactly jog-scale units
# ============================================================

setp joint.0.jog-vel-mode 0
setp joint.1.jog-vel-mode 0
setp joint.3.jog-vel-mode 0

# axis-selected-count already driven by encoder.00.count above
net axis-selected-count =>  joint.0.jog-counts
net axis-selected-count =>  joint.1.jog-counts
net axis-selected-count =>  joint.3.jog-counts
net axis-selected-count =>  axis.x.jog-counts
net axis-selected-count =>  axis.y.jog-counts
net axis-selected-count =>  axis.z.jog-counts

# joint-select-a/b/c already driven by gpio above
net joint-select-a  =>  joint.0.jog-enable
net joint-select-b  =>  joint.1.jog-enable
net joint-select-c  =>  joint.3.jog-enable
net joint-select-a  =>  axis.x.jog-enable
net joint-select-b  =>  axis.y.jog-enable
net joint-select-c  =>  axis.z.jog-enable

# Multiplier: IO47=x1(default)  IO48=x10  IO49=x100
# jog-scale = mm per encoder count (position mode)
# Rotate one click and check how much count changes:
#   1 count/click → use 0.001 / 0.01 / 0.1
#   4 counts/click → use 0.00025 / 0.0025 / 0.025

loadrt mux4 names=mpg-scale-mux
addf mpg-scale-mux  servo-thread

setp mpg-scale-mux.in0  0.001
setp mpg-scale-mux.in1  0.01
setp mpg-scale-mux.in2  0.1
setp mpg-scale-mux.in3  0.001

net mpg-x10   <=  hm2_7i96s.0.gpio.048.in_not
net mpg-x100  <=  hm2_7i96s.0.gpio.049.in_not
net mpg-x10   =>  mpg-scale-mux.sel0
net mpg-x100  =>  mpg-scale-mux.sel1

net mpg-scale <=  mpg-scale-mux.out
net mpg-scale =>  joint.0.jog-scale
net mpg-scale =>  joint.1.jog-scale
net mpg-scale =>  joint.3.jog-scale
net mpg-scale =>  axis.x.jog-scale
net mpg-scale =>  axis.y.jog-scale
net mpg-scale =>  axis.z.jog-scale
